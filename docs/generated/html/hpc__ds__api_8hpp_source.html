919 template <cnpts::Scalar T>
920 void ImageView::read_blocks(const std::vector<i3d::Vector3d<int>>& coords,
921  i3d::Image3d<T>& dest,
922  const std::vector<i3d::Vector3d<int>>& offsets,
923  dataset_props_ptr props /* = nullptr */
924 ) const {
925 
926  if (!props)
927  props = get_properties();
928 
929  if (!details::matches_image_type(dest, props->voxel_type))
930  throw std::logic_error("Server and i3d image type does not match\n");
931 
932  auto resolutions = props->get_all_resolutions();
933  if (std::ranges::find(resolutions, _resolution) == end(resolutions))
934  throw std::logic_error(
935  fmt::format("Resolution {} not supported by server\n",
936  details::to_string(_resolution))
937  .c_str());
938 
939  if (!props->timepoint_ids.contains(_timepoint))
940  throw std::logic_error(
941  fmt::format("Timepoint {} not supported by server\n",
942  details::to_string(_timepoint))
943  .c_str());
944 
945  if (_channel >= props->channels)
946  throw std::logic_error(
947  fmt::format("Channel {} not supported by server\n",
948  details::to_string(_channel))
949  .c_str());
950 
951  if (_angle >= props->angles)
952  throw std::logic_error(fmt::format("Angle {} not supported by server\n",
953  details::to_string(_angle))
954  .c_str());
955 
956  /* Fetched properties from server */
957  std::string dataset_url = details::get_dataset_url(_ip, _port, _uuid);
958  i3d::Vector3d<int> block_dim = props->get_block_dimensions(_resolution);
959 
960  i3d::Vector3d<int> img_dim = props->get_img_dimensions(_resolution);
961 
962  if (coords.size() != offsets.size())
963  throw std::logic_error("Count of coordinates != count of offsets");
964 
965  if (!details::check_block_coords(coords, img_dim, block_dim))
966  throw std::out_of_range("Blocks out of range");
967 
968  /* prepare request url */
969  std::string session_url = details::requests::session_url_request(
970  dataset_url, _resolution, _version);
971 
972  if (session_url.ends_with('/'))
973  session_url.pop_back();
974 
975  std::vector<std::pair<std::string, std::vector<std::size_t>>> requests =
976  details::create_requests(coords, session_url, _timepoint, _channel,
977  _angle);
978 
979  for (const auto& [req, idxs] : requests) {
980  auto [data, response] = details::requests::make_request(req);
981 
982  std::size_t start_i = 0;
983  for (std::size_t i : idxs) {
984  i3d::Vector3d<int> block_size =
985  props->get_block_size(coords[i], _resolution);
986  std::size_t data_size = details::data_manip::get_block_data_size(
987  block_size, props->voxel_type);
988 
989  details::data_manip::read_data(
990  std::span(data.begin() + start_i, data_size), props->voxel_type,
991  dest, offsets[i], block_size);
992 
993  start_i += data_size;
994  }
995  }
996 }

998 template <cnpts::Scalar T>
999 i3d::Image3d<T>
1000 ImageView::read_region(i3d::Vector3d<int> start_point,
1001  i3d::Vector3d<int> end_point,
1002  dataset_props_ptr props /* = nullptr */) const {
1003  if (!props)
1004  props = get_properties();
1005 
1006  i3d::Vector3d img_dim = props->get_img_dimensions(_resolution);
1007  i3d::Vector3d block_dim = props->get_block_dimensions(_resolution);
1008 
1009  std::vector<i3d::Vector3d<int>> coords = details::get_intercepted_blocks(
1010  start_point, end_point, img_dim, block_dim);
1011 
1012  std::vector<i3d::Vector3d<int>> offsets;
1013  for (auto coord : coords)
1014  offsets.emplace_back(coord * block_dim - start_point);
1015 
1016  i3d::Image3d<T> out_img;
1017  out_img.MakeRoom(end_point - start_point);
1018 
1019  read_blocks(coords, out_img, offsets, props);
1020  return out_img;

1095

1097  i3d::Vector3d<int> img_dim = props->get_img_dimensions(_resolution);

1102  throw std::logic_error("Count of coordinates != count of offsets");

1117  134217728 /

1145  }

1169  (img_dim + block_dim - 1) / block_dim; // Ceiling

1173  std::vector<i3d::Vector3d<int>> offsets;

1176  for (int y = 0; y < block_count.y; ++y)
1177  for (int z = 0; z < block_count.z; ++z) {
1178  blocks.emplace_back(x, y, z);
1179  offsets.emplace_back(x * block_dim.x, y * block_dim.y,
1180  z * block_dim.z);
1181  }
